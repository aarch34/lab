import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.datasets import load_iris

# Load the Iris dataset from sklearn
iris = load_iris()

# Create a DataFrame with feature data and column names
df = pd.DataFrame(data=iris.data, columns=iris.feature_names)

# Add target variable (species) as a categorical column
df['species'] = pd.Categorical.from_codes(iris.target, iris.target_names)

# Display the first few rows of the DataFrame
print(df.head())

# Display the 'species' column
print("\nSpecies column:")
print(df['species'].head())

# 1. Select a numerical column (e.g., 'sepal length (cm)')
numerical_column = 'sepal length (cm)'

# 2. Compute statistical values
mean_value = df[numerical_column].mean()
median_value = df[numerical_column].median()
mode_value = df[numerical_column].mode()[0]
std_dev = df[numerical_column].std()
variance = df[numerical_column].var()
data_range = df[numerical_column].max() - df[numerical_column].min()

# 3. Print the computed values
print(f"\nDescriptive Statistics for '{numerical_column}':")
print(f"Mean               : {mean_value:.2f}")
print(f"Median             : {median_value:.2f}")
print(f"Mode               : {mode_value:.2f}")
print(f"Standard Deviation : {std_dev:.2f}")
print(f"Variance           : {variance:.2f}")
print(f"Range              : {data_range:.2f}")
